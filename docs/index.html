Power Query M — Conexão Direta ao Kanboard

No Power BI Desktop: Obter Dados → Consulta em Branco → Editor Avançado

let
  url = "http://kanboard.eblsolucoescorp.tec.br/jsonrpc.php",
  body = "{""jsonrpc"":""2.0"",""method"":""getAllTasks"",""id"":1,""params"":{""project_id"":11,""status_id"":1}}",
  creds = Binary.ToText(Text.ToBinary("admin:Senha@2026"), BinaryEncoding.Base64),
  headers = [#"Content-Type"="application/json", Authorization="Basic " & creds],
  response = Web.Contents(url, [Headers=headers, Content=Text.ToBinary(body)]),
  json = Json.Document(response),
  result = json[result],
  table = Table.FromList(result, Splitter.SplitByNothing())
in table

Medidas DAX Prontas

Total Demandas = COUNTROWS(Demandas)
Implementadas = CALCULATE(COUNTROWS(Demandas), Demandas[status]="Implementado")
Taxa Conclusão % = DIVIDE([Implementadas],[Total Demandas],0)*100
Alta Prioridade = CALCULATE(COUNTROWS(Demandas), Demandas[pri]="Alta")
Em Andamento = CALCULATE(COUNTROWS(Demandas), Demandas[fase] IN {"02. Refinamento","06. Aprovação","07. Desenvolvimento","09. Deploy"})
Valor Total = SUM(Demandas[valor])
Horas Totais = SUM(Demandas[horas])
Backlog = CALCULATE(COUNTROWS(Demandas), Demandas[fase]="01. Backlog")
